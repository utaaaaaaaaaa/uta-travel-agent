### 工具分层

```
┌─────────────────────────────────────────────────────────────┐
│                    Agent 工具配置                            │

│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 信息获取策略

| 信息类型 | 获取方式 | 时效性 |
|----------|----------|--------|
| 历史、文化背景 | Wikipedia API → RAG | 静态 |
| 景点介绍、故事 | Wikipedia API → RAG | 静态 |
| 美食、住宿推荐 | Web Reader → RAG | 半静态 |
| 门票价格、开放时间 | Tavily 实时搜索 | 动态 |
| 天气、交通状况 | Tavily 实时搜索 | 动态 |
| 近期活动、新闻 | News API / Tavily | 动态 |

### 实时搜索触发规则

Destination Agent 判断是否使用实时搜索：

**关键词规则**（优先级高）：
- 时间相关: "今天", "现在", "目前", "本周", "最近"
- 价格相关: "多少钱", "票价", "费用", "价格"
- 状态相关: "开放时间", "营业时间", "关门", "是否开门"
- 天气相关: "天气", "气温", "下雨"
- 交通相关: "怎么去", "交通", "路线", "拥堵"

**LLM 判断**（补充）：
- 当 RAG 结果明显过时或不包含用户需要的信息时

### 2.1 移除 simulateProgress

**当前问题** (router.go:564-608):


**改造方案**:

```go
func (r *Router) executeAgentCreation(task *agent.AgentTask, ag *agent.DestinationAgent) {
    ctx := context.Background()
    startTime := time.Now()

    // 更新任务状态
    task.Status = agent.TaskStatusRunning
    r.scheduler.Save(task)

    // 1. 创建 Researcher LLMAgent
    researcher := agent.NewLLMAgent(agent.LLMAgentConfig{
        AgentType:   agent.AgentTypeResearcher,
        LLMProvider: r.llmProvider,
        SystemPrompt: prompts.GetResearcherPrompt(ag.Destination, ag.Theme),
        Tools:       r.toolRegistry,
        MaxSteps:    15,
    })

    // 2. 执行研究任务
    goal := fmt.Sprintf("收集 %s 的旅游信息，主题: %s", ag.Destination, ag.Theme)
    result, err := researcher.Run(ctx, goal)
    if err != nil {
        task.Status = agent.TaskStatusFailed
        task.Error = err.Error()
        r.scheduler.Save(task)
        return
    }

    // 3. 更新探索日志到任务
    task.ExplorationLog = researcher.GetExplorationLog()
    r.scheduler.Save(task)

    // 4. 创建向量索引
    collectionID, err := r.createVectorIndex(ctx, result.Documents)
    if err != nil {
        task.Status = agent.TaskStatusFailed
        task.Error = fmt.Sprintf("创建索引失败: %v", err)
        r.scheduler.Save(task)
        return
    }

    // 5. 完成
    task.Status = agent.TaskStatusCompleted
    task.Result = map[string]any{
        "collection_id":  collectionID,
        "document_count": len(result.Documents),
    }
    ag.Status = agent.StatusReady
    ag.VectorCollectionID = collectionID
    ag.DocumentCount = len(result.Documents)

    r.scheduler.Save(task)
    r.registry.Update(ag)
}
```

### 2.2 实现向量索引创建

```go
func (r *Router) createVectorIndex(ctx context.Context, documents []Document) (string, error) {
    collectionID := fmt.Sprintf("dest_%d", time.Now().UnixNano())

    // 1. 创建 Qdrant Collection
    err := r.qdrantClient.CreateCollection(ctx, collectionID, 384) // embedding 维度
    if err != nil {
        return "", err
    }

    // 2. 生成 Embedding
    texts := make([]string, len(documents))
    for i, doc := range documents {
        texts[i] = doc.Content
    }

    embeddings, err := r.embeddingClient.BatchEmbed(ctx, texts, "")
    if err != nil {
        return "", err
    }

    // 3. 写入 Qdrant
    points := make([]qdrant.Point, len(documents))
    for i, doc := range documents {
        points[i] = qdrant.Point{
            ID:      uuid.New().String(),
            Vector:  embeddings[i],
            Payload: map[string]any{
                "content":  doc.Content,
                "url":      doc.URL,
                "title":    doc.Title,
                "source":   doc.Source,
            },
        }
    }

    err = r.qdrantClient.Upsert(ctx, collectionID, points)
    if err != nil {
        return "", err
    }

    return collectionID, nil
}
```

```go
// 在 Researcher 的 System Prompt 中注入主题关键词
func GetResearcherPrompt(destination, theme string) string {
    themeKeywords := map[string][]string{
        "cultural": {"历史", "文化", "寺庙", "博物馆", "古迹", "世界遗产"},
        "food":     {"美食", "餐厅", "小吃", "料理", "市场", "特色菜"},
        "adventure":{"户外", "徒步", "探险", "自然", "公园", "山川"},
        "art":      {"美术馆", "艺术", "画廊", "设计", "展览"},
    }

    keywords := themeKeywords[theme]

    return fmt.Sprintf(`
你是专业的旅游研究员。正在研究目的地: %s
主题: %s

请围绕以下关键词进行搜索:
%s

搜索策略:
1. 先搜索基本信息 (概况、历史)
2. 再搜索主题相关信息 (%s)
3. 搜索实用信息 (交通、住宿)
4. 搜索小众推荐

使用 brave_search 工具进行搜索，使用 web_reader 读取网页内容。
`, destination, theme, strings.Join(keywords, ", "), theme)
}
```

### 3.2 景点列表 API

```go
// GET /api/v1/agents/{id}/attractions
func (r *Router) handleAgentAttractions(w http.ResponseWriter, req *http.Request) {
    agentID := req.PathValue("id")
    ag, exists := r.registry.Get(agentID)
    if !exists {
        writeError(w, http.StatusNotFound, "agent not found")
        return
    }

    // 从 Qdrant 检索景点
    attractions, err := r.ragSvc.GetAttractions(ctx, ag.VectorCollectionID)
    if err != nil {
        writeError(w, http.StatusInternalServerError, err.Error())
        return
    }

    writeJSON(w, http.StatusOK, map[string]any{
        "attractions": attractions,
    })
}
```

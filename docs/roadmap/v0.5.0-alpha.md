1. **搜索工具** 返回硬编码结果 → 无法收集真实信息

4. **Agent 创建流程** 使用 simulateProgress → 没有真实执行

```go
// internal/tools/brave_search.go
package tools

import (
    "context"
    "encoding/json"
    "fmt"
    "net/http"
    "net/url"
)

type BraveSearchTool struct {
    apiKey string
    client *http.Client
}

func NewBraveSearchTool(apiKey string) *BraveSearchTool {
    return &BraveSearchTool{
        apiKey: apiKey,
        client: &http.Client{Timeout: 10 * time.Second},
    }
}

func (t *BraveSearchTool) Execute(ctx context.Context, params map[string]any) (*ToolResult, error) {
    query, ok := params["query"].(string)
    if !ok {
        return nil, fmt.Errorf("query parameter required")
    }

    // 调用 Brave Search API
    reqURL := fmt.Sprintf(
        "https://api.search.brave.com/res/v1/web/search?q=%s&count=10",
        url.QueryEscape(query),
    )

    req, _ := http.NewRequestWithContext(ctx, "GET", reqURL, nil)
    req.Header.Set("Accept", "application/json")
    req.Header.Set("X-Subscription-Token", t.apiKey)

    resp, err := t.client.Do(req)
    if err != nil {
        return nil, fmt.Errorf("brave search failed: %w", err)
    }
    defer resp.Body.Close()

    var result BraveSearchResponse
    if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
        return nil, fmt.Errorf("parse response failed: %w", err)
    }

    // 转换为 ToolResult
    return &ToolResult{
        Success: true,
        Data: map[string]any{
            "query":   query,
            "results": t.extractResults(result),
        },
    }, nil
}
```

```go
// internal/tools/web_reader.go
package tools

import (
    "context"
    "fmt"
    "io"
    "net/http"
    "strings"

    "github.com/PuerkitoBio/goquery"
)

type WebReaderTool struct {
    client *http.Client
}

func NewWebReaderTool() *WebReaderTool {
    return &WebReaderTool{
        client: &http.Client{
            Timeout: 15 * time.Second,
        },
    }
}

func (t *WebReaderTool) Execute(ctx context.Context, params map[string]any) (*ToolResult, error) {
    urlStr, ok := params["url"].(string)
    if !ok {
        return nil, fmt.Errorf("url parameter required")
    }

    req, _ := http.NewRequestWithContext(ctx, "GET", urlStr, nil)
    req.Header.Set("User-Agent", "Mozilla/5.0 (compatible; UTA-Bot/1.0)")

    resp, err := t.client.Do(req)
    if err != nil {
        return nil, fmt.Errorf("fetch url failed: %w", err)
    }
    defer resp.Body.Close()

    // 解析 HTML 提取正文
    doc, err := goquery.NewDocumentFromReader(resp.Body)
    if err != nil {
        return nil, fmt.Errorf("parse html failed: %w", err)
    }

    content := t.extractContent(doc)

    return &ToolResult{
        Success: true,
        Data: map[string]any{
            "url":     urlStr,
            "title":   doc.Find("title").Text(),
            "content": content,
        },
    }, nil
}

func (t *WebReaderTool) extractContent(doc *goquery.Document) string {
    // 移除无关元素
    doc.Find("script, style, nav, header, footer, aside, .ads, .sidebar").Remove()

    // 优先提取 article 或 main
    var content string
    if article := doc.Find("article").First(); article.Length() > 0 {
        content = article.Text()
    } else if main := doc.Find("main").First(); main.Length() > 0 {
        content = main.Text()
    } else {
        content = doc.Find("body").Text()
    }

    // 清理空白
    return strings.TrimSpace(content)
}
```

```go
func (r *Router) executeAgentCreation(task *agent.AgentTask, ag *agent.DestinationAgent) {
    ctx := context.Background()
    startTime := time.Now()

    // 更新任务状态
    task.Status = agent.TaskStatusRunning
    r.scheduler.Save(task)

    // 1. 创建 Researcher LLMAgent
    researcher := agent.NewLLMAgent(agent.LLMAgentConfig{
        AgentType:   agent.AgentTypeResearcher,
        LLMProvider: r.llmProvider,
        SystemPrompt: prompts.GetResearcherPrompt(ag.Destination, ag.Theme),
        Tools:       r.toolRegistry,
        MaxSteps:    15,
    })

    // 2. 执行研究任务
    goal := fmt.Sprintf("收集 %s 的旅游信息，主题: %s", ag.Destination, ag.Theme)
    result, err := researcher.Run(ctx, goal)
    if err != nil {
        task.Status = agent.TaskStatusFailed
        task.Error = err.Error()
        r.scheduler.Save(task)
        return
    }

    // 3. 更新探索日志到任务
    task.ExplorationLog = researcher.GetExplorationLog()
    r.scheduler.Save(task)

    // 4. 创建向量索引
    collectionID, err := r.createVectorIndex(ctx, result.Documents)
    if err != nil {
        task.Status = agent.TaskStatusFailed
        task.Error = fmt.Sprintf("创建索引失败: %v", err)
        r.scheduler.Save(task)
        return
    }

    // 5. 完成
    task.Status = agent.TaskStatusCompleted
    task.Result = map[string]any{
        "collection_id":  collectionID,
        "document_count": len(result.Documents),
    }
    ag.Status = agent.StatusReady
    ag.VectorCollectionID = collectionID
    ag.DocumentCount = len(result.Documents)

    r.scheduler.Save(task)
    r.registry.Update(ag)
}
```

### 2.2 实现向量索引创建

```go
func (r *Router) createVectorIndex(ctx context.Context, documents []Document) (string, error) {
    collectionID := fmt.Sprintf("dest_%d", time.Now().UnixNano())

    // 1. 创建 Qdrant Collection
    err := r.qdrantClient.CreateCollection(ctx, collectionID, 384) // embedding 维度
    if err != nil {
        return "", err
    }

    // 2. 生成 Embedding
    texts := make([]string, len(documents))
    for i, doc := range documents {
        texts[i] = doc.Content
    }

    embeddings, err := r.embeddingClient.BatchEmbed(ctx, texts, "")
    if err != nil {
        return "", err
    }

    // 3. 写入 Qdrant
    points := make([]qdrant.Point, len(documents))
    for i, doc := range documents {
        points[i] = qdrant.Point{
            ID:      uuid.New().String(),
            Vector:  embeddings[i],
            Payload: map[string]any{
                "content":  doc.Content,
                "url":      doc.URL,
                "title":    doc.Title,
                "source":   doc.Source,
            },
        }
    }

    err = r.qdrantClient.Upsert(ctx, collectionID, points)
    if err != nil {
        return "", err
    }

    return collectionID, nil
}
```

```go
// GET /api/v1/agents/{id}/attractions
func (r *Router) handleAgentAttractions(w http.ResponseWriter, req *http.Request) {
    agentID := req.PathValue("id")
    ag, exists := r.registry.Get(agentID)
    if !exists {
        writeError(w, http.StatusNotFound, "agent not found")
        return
    }

    // 从 Qdrant 检索景点
    attractions, err := r.ragSvc.GetAttractions(ctx, ag.VectorCollectionID)
    if err != nil {
        writeError(w, http.StatusInternalServerError, err.Error())
        return
    }

    writeJSON(w, http.StatusOK, map[string]any{
        "attractions": attractions,
    })
}
```

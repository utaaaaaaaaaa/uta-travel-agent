### Phase 3: Agent Refactoring (Week 2-3)

**Goal**: Refactor MainAgent and create dedicated GuideAgent

#### 3.1 Refactored MainAgent

```go
// internal/agent/main_agent.go
type MainAgent struct {
    *BaseSessionAgent

    subagents   map[AgentType]Agent
}

func (a *MainAgent) Chat(ctx context.Context, message string) (string, error) {
    // 1. Update session
    a.session.Touch()

    // 2. Add to memory
    a.memory.AddMessage("user", message)

    // 3. Build context
    messages := a.contextEngine.BuildContext(a.memory, a.contextEngine.maxTokens)

    // 4. Call LLM
    response, _ := a.llmProvider.CompleteWithSystem(ctx, a.systemPrompt, messages)

    // 5. Save response
    a.memory.AddMessage("assistant", response.Content)

    // 6. Auto-save session
    a.SaveSession()

    return response.Content, nil
}
```

#### 3.2 Dedicated GuideAgent

```go
// internal/agent/guide_agent.go
type GuideAgent struct {
    *BaseSessionAgent

    destination  string
    collectionID string
    ragService   *rag.Service
}

func (a *GuideAgent) Guide(ctx context.Context, query string) (string, error) {
    // 1. Update session
    a.session.Touch()

    // 2. RAG query
    context, sources := a.ragService.Query(ctx, a.collectionID, query)

    // 3. Build context with RAG results
    messages := a.contextEngine.BuildContext(a.memory, a.contextEngine.maxTokens)
    messages = append(messages, llm.Message{
        Role: "system",
        Content: fmt.Sprintf("Reference knowledge:\n%s", context),
    })

    // 4. Call LLM
    response, _ := a.llmProvider.CompleteWithSystem(ctx, a.systemPrompt, messages)

    // 5. Remember important info
    a.memory.AddMessage("user", query)
    a.memory.AddMessage("assistant", response.Content)

    return response.Content, nil
}
```

### Phase 5: Context Engineering (Week 4)

**Goal**: Implement smart context management

#### 5.1 Token Estimation

```go
func (e *ContextEngineer) EstimateTokens(content string) int {
    // Rough estimation: ~4 chars per token for Chinese
    // ~0.75 words per token for English
    return len(content) / 4
}
```

#### 5.2 Context Prioritization

```go
type Priority int

const (
    PriorityCritical Priority = 100  // Current query, system prompt
    PriorityHigh     Priority = 75   // Recent messages
    PriorityMedium   Priority = 50   // Summaries
    PriorityLow      Priority = 25   // Old messages
)

func (e *ContextEngineer) BuildContext(memory *AgentMemory, maxTokens int) []llm.Message {
    items := memory.GetAll()

    // Sort by priority and recency
    sort.Slice(items, func(i, j int) bool {
        return e.getPriority(items[i]) > e.getPriority(items[j])
    })

    // Build messages until token limit
    var messages []llm.Message
    currentTokens := 0

    for _, item := range items {
        tokens := e.EstimateTokens(item.Content)
        if currentTokens + tokens > maxTokens {
            // Try compression
            compressed := e.Compress([]MemoryItem{item})
            if e.EstimateTokens(compressed) < tokens {
                messages = append(messages, llm.Message{
                    Role: "system",
                    Content: compressed,
                })
                break
            }
        }

        messages = append(messages, llm.Message{
            Role: item.Metadata["role"].(string),
            Content: item.Content,
        })
        currentTokens += tokens
    }

    return messages
}
```

#### 5.3 Context Compression

```go
func (e *ContextEngineer) Compress(items []MemoryItem) string {
    // Use LLM to summarize
    prompt := "Summarize the following conversation in 2-3 sentences:\n\n"
    for _, item := range items {
        prompt += fmt.Sprintf("%s: %s\n", item.Metadata["role"], item.Content)
    }

    summary, _ := e.llmProvider.Complete(context.Background(), []llm.Message{
        {Role: "user", Content: prompt},
    })

    return summary.Content
}
```

## Risks & Mitigation

| Risk | Mitigation |
|------|------------|
| Context compression quality | Fallback to truncation if compression fails |
| Session storage performance | Index on user_id, state, last_active_at |
| Memory overhead | Lazy loading, pagination |
| Frontend complexity | Reuse ChatGPT UI patterns |

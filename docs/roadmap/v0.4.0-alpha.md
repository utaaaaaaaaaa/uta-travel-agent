#### 2.2 MainAgent 真正编排 Subagent

```go
func (a *MainAgent) CreateDestinationAgent(ctx context.Context, req CreateRequest) (*AgentTask, error) {
    task := &AgentTask{
        ID:      generateID(),
        AgentID: req.Destination,
        Status:  "running",
        Goal:    fmt.Sprintf("创建 %s 导游 Agent", req.Destination),
    }

    // 保存任务
    a.repo.SaveTask(ctx, task)

    // 启动后台执行
    go a.executeCreationTask(context.Background(), task, req)

    return task, nil
}

func (a *MainAgent) executeCreationTask(ctx context.Context, task *AgentTask, req CreateRequest) {
    defer func() {
        task.CompletedAt = &time.Time{}
        a.repo.UpdateTask(ctx, task)
    }()

    // Step 1: Researcher
    a.reportProgress(task, "researching", 20, "正在搜索信息...")
    result, err := a.researcher.Run(ctx, req.Destination)
    if err != nil {
        task.Status = "failed"
        task.Error = err.Error()
        return
    }
    documents := result.Output["documents"]

    // Step 2: Curator
    a.reportProgress(task, "curating", 50, "正在整理信息...")
    result, err = a.curator.Run(ctx, documents)
    if err != nil {
        task.Status = "failed"
        task.Error = err.Error()
        return
    }
    curatedDocs := result.Output["curated"]

    // Step 3: Indexer
    a.reportProgress(task, "indexing", 80, "正在构建向量索引...")
    result, err = a.indexer.Run(ctx, curatedDocs)
    if err != nil {
        task.Status = "failed"
        task.Error = err.Error()
        return
    }
    collectionID := result.Output["collection_id"]

    // 完成
    task.Status = "completed"
    task.Result = map[string]any{
        "collection_id": collectionID,
        "document_count": len(curatedDocs),
    }

    // 更新 Agent 状态
    agent, _ := a.repo.GetAgent(ctx, task.AgentID)
    agent.Status = "ready"
    agent.VectorCollectionID = collectionID
    a.repo.UpdateAgent(ctx, agent)
}
```

const handleCreate = async () => {
  setStatus('creating');

  // 1. 创建 Agent (返回 agent_id 和 task_id)
  const response = await fetch("/api/v1/agents", {
    method: "POST",
    body: JSON.stringify({ destination, theme, languages }),

  const { agent_id, task_id } = await response.json();

  // 2. 监听 SSE 进度
  const eventSource = new EventSource(`/api/v1/tasks/${task_id}/stream`);

  eventSource.onmessage = (event) => {
    const data = JSON.parse(event.data);

    if (data.stage) {
      // 更新进度
      updateProgress(data);
    }

    if (data.exploration_step) {
      // 添加探索点到雷达图
      addExplorationPoint(data.exploration_step);
    }
  };

  eventSource.addEventListener('complete', (event) => {
    const data = JSON.parse(event.data);
    setAgentId(data.agent_id);
    setStatus('completed');
    eventSource.close();
  });
};

### Phase 3: Subagent 升级为真正的 Agent (Week 3)

#### 3.1 当前 vs 目标

| 组件 | 当前 (ResearcherAgent) | 目标 (使用 LLMAgent) |
|------|----------------------|---------------------|
| **LLM Brain** | ❌ 无 | ✅ 每步调用 LLM 决策 |
| **Memory** | ✅ 有 | ✅ 独立记忆 |
| **Prompt** | ❌ 无角色定义 | ✅ 定义"旅游研究员"角色 |
| **Action Flow** | ❌ 硬编码 3 步 | ✅ ReAct 循环 |
| **自主探索** | ❌ 固定搜索 | ✅ 可自主发现新方向 |

```go
func (a *GuideAgent) Run(ctx context.Context, query string) (*AgentResult, error) {
    // 1. RAG 检索相关上下文
    context, err := a.ragQuery(ctx, query, a.collectionID)
    if err != nil {
        return nil, err
    }

    // 2. LLM 生成讲解
    prompt := fmt.Sprintf(`
        作为 %s 的专业导游，请基于以下知识回答用户问题。

        相关知识：
        %s

        用户问题：%s

        请用生动有趣的方式讲解，包含历史背景、文化内涵、游览建议。
    `, a.destination, context, query)

    response, err := a.llm.Complete(ctx, prompt)

    return &AgentResult{
        Output: map[string]any{
            "response": response,
            "sources":  context.Sources,
        },
    }, nil
}
```

func (r *Router) handleGuideChat(w http.ResponseWriter, req *http.Request) {
    agentID := req.PathValue("id")

    // 设置 SSE
    w.Header().Set("Content-Type", "text/event-stream")
    w.Header().Set("Cache-Control", "no-cache")

    // 获取 Agent
    agent, _ := r.registry.Get(agentID)
    guide := agent.GuideAgent

    // 流式生成
    ch, _ := guide.ChatStream(ctx, query)
    for chunk := range ch {
        fmt.Fprintf(w, "data: %s\n\n", chunk)
        w.(http.Flusher).Flush()
    }
}
